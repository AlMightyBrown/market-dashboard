async function fetchQuote(symbol) {

  if (!useLive) return null;

  try {

    const r = await fetch(`https://finnhub.io/api/v1/quote?symbol=${symbol}&token=${API_KEY}`);

    const d = await r.json();

    if (d.c) return { price: d.c, change: d.d, changePercent: d.dp, high: d.h, low: d.l, prevClose: d.pc };

  } catch(e) { console.warn('Fetch error', symbol, e); }

  return null;

}


function demoData(basePrice, volatility) {

  const change = (Math.random() - 0.48) * volatility;

  const pct = (change / basePrice) * 100;

  return { price: basePrice + change, change, changePercent: pct };

}

function renderCard(name, data, prefix = '') {

  const isUp = data.change >= 0;

  const arrow = isUp ? '▲' : '▼';

  const cls = isUp ? 'up' : 'down';

  const priceStr = data.price > 1000 ? data.price.toLocaleString('en-US', {minimumFractionDigits:2, maximumFractionDigits:2}) :

                   data.price < 10 ? data.price.toFixed(4) : data.price.toFixed(2);

  return `<div class="card">

    <div class="card-name">${prefix}${name}</div>

    <div class="card-price">${priceStr}</div>

    <div class="card-change ${cls}">${arrow} ${Math.abs(data.change).toFixed(2)} (${Math.abs(data.changePercent).toFixed(2)}%)</div>

  </div>`;

}


async function loadIndices() {

  const grid = document.getElementById('indicesGrid');

  let html = '';

  for (const idx of INDICES) {

    let data = await fetchQuote(idx.finnhub);

    if (!data) data = demoData(DEMO_PRICES[idx.finnhub], DEMO_PRICES[idx.finnhub] * 0.012);

    html += renderCard(idx.name, data);

  }

  grid.innerHTML = html;

}


async function loadForex() {

  const grid = document.getElementById('forexGrid');

  let html = '';

  for (const fx of FOREX) {

    let data = await fetchQuote(fx.symbol);

    if (!data) {

      const bp = DEMO_PRICES[fx.display];

      data = demoData(bp, bp * 0.005);

    }

    html += renderCard(fx.display, data);

  }

  grid.innerHTML = html;

}

async function loadSectors() {

  const heatmap = document.getElementById('heatmap');

  let html = '';

  for (const s of SECTORS) {

    let data = await fetchQuote(s.etf);

    if (!data) data = demoData(DEMO_PRICES[s.etf], DEMO_PRICES[s.etf] * 0.015);

    const pct = data.changePercent;

    const intensity = Math.min(Math.abs(pct) / 2, 1);

    const bg = pct >= 0 ? `rgba(34,197,94,${0.15 + intensity * 0.45})` : `rgba(239,68,68,${0.15 + intensity * 0.45})`;

    html += `<div class="heat-cell" style="background:${bg}"><div class="label">${s.name}</div><div class="val">${pct >= 0 ? '+' : ''}${pct.toFixed(2)}%</div></div>`;

  }

  heatmap.innerHTML = html;

}


function loadTicker() {

  const tape = document.getElementById('tickerTape');

  const items = [...INDICES, ...FOREX.map(f => ({name: f.display, finnhub: f.display})), ...CRYPTO.map(c => ({name: c.display, finnhub: c.display}))];

  let html = '';

  for (let rep = 0; rep < 2; rep++) {

    for (const item of items) {

      const bp = DEMO_PRICES[item.finnhub] || 100;

      const d = demoData(bp, bp * 0.01);

      const isUp = d.change >= 0;

      const color = isUp ? 'var(--green)' : 'var(--red)';

      const arrow = isUp ? '▲' : '▼';

      html += `<div class="ticker-item"><span class="sym">${item.name}</span> <span style="color:${color}">${arrow} ${Math.abs(d.changePercent).toFixed(2)}%</span></div>`;

    }

  }

  tape.innerHTML = html;

}
